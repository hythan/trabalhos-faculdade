Objetos

Um objeto é um tipo especial de variável que contém propriedades e métodos.
A criação de objetos (classes) em Javascript se da por meio de uma função construtora ou notação literal.

Declaração Objeto Array String

Função Contrutora

Nessa forma de declaração, o nome da função deve ser inicado com letra maiúscula.

var Pessoa = function (nome, sobrenome, email) {
	'use strict';
	this.nome = nome;
	this.sobrenome = sobrenome;
	this.email = email;
	this.imprime = 'Nome da pessoa: ' +this.nome + '
Sobrenome: ' + this.sobrenome + '
Email: ' + this.email;
};
var pessoa = new Pessoa('João', 'Souza', 'joão_123@gmail.com');

                                

                                
Notação Literal

                                
var pai = {
nome: 'Pedro',
sobrenome: 'Pato',
email: 'pedro_pato@gmail.com',
imprime: function () {
	'use strict';
return 'Nome da pessoa: ' + this.nome + '
Sobrenome: ' + this.sobrenome + '
Email: ' + this.email;
	}
};

var filho = Object.create(pai);
filho.nome = 'Augusto';
filho.email = 'augustopato123@gmail.com';
filho.imprime();

                                

                            




                            

                                
Array

                                
Array é uma lista indexada de elementos, a qual é possível armazenar mais de um elemento.
A criação de um array em javascript se da pela notação 'var vetor = []' ou 'var vetor = new Array()'.

                                
var vetor = [];

Array de Objetos

                                
Exemplo, preencha os campos do formulário e clique em cadastrar.

                                
var ALUNOS = [];
var DISCIPLINAS = [];

function Aluno(ra, nome) {
	this.ra = ra;
	this.nome = nome;
	this.diciplinas = [];
}

var disciplina = {
	nome: ''
};
$('#bt-adicionar-disciplina').addEventListener('click', function () {

	var materia = Object.create(disciplina);
	materia.nome = document.forms[0].elements[2].value;
	document.forms[0].elements[2].value = '';
	DISCIPLINAS.push(materia);
});
$('#bt-cadastrar-aluno').onclick = function () {

	var aluno = new Aluno(document.forms[0].elements[0].value, document.forms[0].elements[1].value);

	for (i in DICIPLINAS) {
		aluno.disciplina.push(DISCIPLINAS[i]);
	}
	ALUNOS.push(aluno);
}

     Strings são úteis para guardar dados que podem ser representados em forma de texto.
